### Boolean

```typescript
const isWaiting: boolean = false
```

### Numbers

```typescript
const age: number = 4
```

### String

```typescript
const greeting: string = 'Bonjour'
```

### Array

```typescript
const numbers: Array<number> = [1, 2, 3, 4, 5]
const numbers: number[] = [1, 2, 3, 4, 5]
```
